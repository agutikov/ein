## Goal

Implement `LatticeDumper` with the per-set + per-entering
filesystem layout designed in
[T1.5a.20.6](../p1.5a_zebra_solution/s1.5a.20_branch_isolation_rearch.md#t15a206--per-child-branch-dump-branchresult--integrate)
and made concrete in
[`algorithm_layer_n.md` § 3d](algorithm_layer_n.md).

After this stage:

```
dump/<puzzle>-<ts>/
  00_root_initial.ein
  00_timeline.jsonl
  layer_summaries/
      layer_01.json
      layer_02.json
      ...
  sets/
      <canonical-slug>/          ← per-SetNode folder; one per dedup-distinct node
          set_info.json          ← canonical_set, labels, parents, state_hash, verdict
          kb_snapshot.ein        ← post-saturation kb
          firings.jsonl          ← saturator firings
          enterings/
              s001_<label-slug>/ ← per-entering (per-label) subfolder
                  arrival.json   ← {canonical_set, parent_set, layer, ts_ms, dedup="miss"|"hit"}
                  # if alive:
                  unconditional_facts.json
                  # if dead:
                  unsat_core.json
                  nogood_emitted.json
              s002_<label-slug>/ ← subsequent entering that merged here
                  arrival.json   ← dedup="hit"
  proof/
      lattice_proof.json         ← S1.5b.29 fills this
  summary.json                   ← top-level run summary
```

The structure is more elaborate than the monotonic dumper
because the lattice has per-set state worth recording.

## `LatticeDumper` surface

`ein.py/src/ein_bot/inference/lattice/state_dump.py` filled in:

```python
"""LatticeDumper — per-set + per-entering filesystem
snapshots.

Reads the Lattice's SetNode store + merged_events to
write a self-contained dump that P1.6's NL renderer can
consume + a human reader can scrub through.
"""
from __future__ import annotations

import json
import time
from dataclasses import dataclass, field, fields
from pathlib import Path
from typing import IO, Any

from ein_bot.inference.apriori import CanonicalSetId
from ein_bot.inference.lattice.lattice import Lattice, SetNode
from ein_bot.inference.set_branch import SetResult
from ein_bot.inference.state_dump import (
    _kb_to_ein_text, _fact_summary, _classify_resat_write,
)
from ein_bot.kb.store import KnowledgeBase

def canonical_slug(C: CanonicalSetId) -> str:
    """`((rel, args), …)` → human-readable folder name."""
    parts: list[str] = []
    for (rn, args) in C:
        chunk = rn + "_" + "-".join(str(a) for a in args)
        parts.append(chunk.replace("_", "-").lower())
    return "_".join(parts) or "root"

@dataclass
class LatticeDumper:
    out_dir: Path
    started_at: float = field(default_factory=time.time)
    _timeline_fp: IO[str] | None = field(default=None, init=False, repr=False)
    _seq: int = field(default=0, init=False)
    # Per-set entering counters (`s001`, `s002`, …).
    _entering_counter: dict[CanonicalSetId, int] = field(default_factory=dict, init=False)

    def __post_init__(self) -> None:
        self.out_dir.mkdir(parents=True, exist_ok=True)
        for sub in ("sets", "layer_summaries", "proof"):
            (self.out_dir / sub).mkdir(exist_ok=True)
        self._timeline_fp = (self.out_dir / "00_timeline.jsonl").open("w")

    # ── Lifecycle hooks ──────────────────────────────────

    def root_initial(self, kb: KnowledgeBase) -> None:
        (self.out_dir / "00_root_initial.ein").write_text(
            _kb_to_ein_text(kb),
        )
        # Root SetNode (empty commitment) lands as a special
        # set folder.
        set_dir = self.out_dir / "sets" / "root"
        set_dir.mkdir(exist_ok=True)
        (set_dir / "kb_snapshot.ein").write_text(
            _kb_to_ein_text(kb),
        )
        self._emit_timeline("root_initial", facts=len(kb.facts))

    def layer_start(self, layer: int, candidates: int) -> None:
        self._emit_timeline(
            "layer_start", layer=layer, candidates=candidates,
        )

    def set_explored(
        self,
        layer: int,
        canonical_set: CanonicalSetId,
        result: SetResult,
        node: SetNode,
        merge_target: SetNode | None,
    ) -> None:
        """Called once per try_set outcome.

        merge_target is the existing node if state-hash dedup
        fired; otherwise None (fresh node).
        """
        # Determine the target set folder (where this entering
        # is recorded).
        if merge_target is not None:
            target_set = merge_target.canonical_set
            dedup = "hit"
        else:
            target_set = canonical_set
            dedup = "miss"

        slug = canonical_slug(target_set)
        set_dir = self.out_dir / "sets" / slug
        set_dir.mkdir(exist_ok=True)

        # First arrival writes the set's main artefacts.
        if dedup == "miss":
            self._write_set_info(set_dir, node)
            (set_dir / "kb_snapshot.ein").write_text(
                _kb_to_ein_text(node.kb_snapshot),
            )
            with (set_dir / "firings.jsonl").open("w") as fp:
                for f in node.firings:
                    fp.write(json.dumps({
                        "rule": f.rule,
                        "redundant": f.redundant,
                        "derived": _fact_summary(f.derived),
                    }) + "\n")

        # Every entering gets an enterings/sNNN_… subfolder.
        n = self._entering_counter.get(target_set, 0) + 1
        self._entering_counter[target_set] = n
        entering_slug = canonical_slug(canonical_set)
        entering_dir = set_dir / "enterings" / f"s{n:03d}_{entering_slug}"
        entering_dir.mkdir(parents=True, exist_ok=True)

        (entering_dir / "arrival.json").write_text(json.dumps({
            "canonical_set": list(canonical_set),
            "parent_node":   list(target_set),
            "layer":         layer,
            "ts_ms":         round((time.time() - self.started_at) * 1000, 3),
            "dedup":         dedup,
            "state_hash":    node.state_hash,
        }, indent=2, default=str))

        if result.kind == "alive":
            (entering_dir / "unconditional_facts.json").write_text(
                json.dumps([
                    _fact_summary(f) for f in result.unconditional_facts
                ], indent=2),
            )
        else:
            (entering_dir / "unsat_core.json").write_text(
                json.dumps([
                    _fact_summary(f) for f in result.unsat_core
                ], indent=2),
            )
            (entering_dir / "nogood_emitted.json").write_text(
                json.dumps({
                    "clause": [list(elem) for elem in canonical_set],
                    "size":   len(canonical_set),
                }, indent=2),
            )

        self._emit_timeline(
            "set_explored", layer=layer,
            canonical_set=list(canonical_set),
            dedup=dedup, kind=result.kind,
            facts_unconditional=len(result.unconditional_facts),
            unsat_core_size=len(result.unsat_core),
            target_set=list(target_set),
        )

    def merge_event(
        self, layer: int, new_label: CanonicalSetId,
        into_node: CanonicalSetId, state_hash: int,
    ) -> None:
        """State-hash dedup merge — auxiliary event recorded
        beyond `set_explored`'s dedup="hit" flag."""
        self._emit_timeline(
            "merge", layer=layer,
            new_label=list(new_label),
            into_node=list(into_node),
            state_hash=state_hash,
        )

    def layer_end(self, layer: int, lattice: Lattice) -> None:
        (self.out_dir / "layer_summaries" / f"layer_{layer:02d}.json").write_text(
            json.dumps({
                "layer":         layer,
                "alive_count":   len(lattice.alive_by_layer[layer]) if layer < len(lattice.alive_by_layer) else 0,
                "dead_count":    len(lattice.dead_by_layer[layer]) if layer < len(lattice.dead_by_layer) else 0,
                "alive_sets":    [list(s) for s in lattice.alive_by_layer[layer]] if layer < len(lattice.alive_by_layer) else [],
                "dead_sets":     [list(s) for s in lattice.dead_by_layer[layer]] if layer < len(lattice.dead_by_layer) else [],
            }, indent=2, default=str),
        )
        self._emit_timeline("layer_end", layer=layer)

    def summary(self, verdict: Any, stats: Any) -> None:
        elapsed = time.time() - self.started_at
        try:
            stats_dict = {f.name: getattr(stats, f.name) for f in fields(stats)}
        except TypeError:
            stats_dict = {"repr": repr(stats)}
        (self.out_dir / "summary.json").write_text(
            json.dumps({
                "verdict": type(verdict).__name__,
                "elapsed_seconds": round(elapsed, 3),
                "stats": stats_dict,
            }, indent=2, sort_keys=True, default=str),
        )
        self._emit_timeline(
            "summary", verdict=type(verdict).__name__,
            elapsed_seconds=round(elapsed, 3),
        )
        if self._timeline_fp is not None:
            self._timeline_fp.close()
            self._timeline_fp = None

    # ── Internals ────────────────────────────────────────

    def _write_set_info(self, set_dir: Path, node: SetNode) -> None:
        (set_dir / "set_info.json").write_text(json.dumps({
            "canonical_set": list(node.canonical_set),
            "labels":        [list(L) for L in node.labels],
            "parents":       [list(P) for P in node.parents],
            "state_hash":    node.state_hash,
            "verdict":       node.verdict,
        }, indent=2, default=str))

    def _emit_timeline(self, event: str, **fields: Any) -> None:
        if self._timeline_fp is None:
            return
        rec = {
            "seq": self._seq,
            "ts_ms": round((time.time() - self.started_at) * 1000, 3),
            "event": event,
            **fields,
        }
        self._timeline_fp.write(json.dumps(rec) + "\n")
        self._timeline_fp.flush()
        self._seq += 1
```

Total ~250 LOC.

## Backbone hook wiring

`lattice_solve` accepts an optional `dumper: LatticeDumper | None = None` parameter and calls hooks at:

| backbone site                                          | hook                                                  |
|--------------------------------------------------------|-------------------------------------------------------|
| right after `Saturator(root_kb).saturate()` in Phase 1 | `dumper.root_initial(root_kb)`                         |
| at the top of each `for layer in …:`                   | `dumper.layer_start(layer, len(candidates))`          |
| after each `try_set` + `lattice.add` outcome          | `dumper.set_explored(layer, C, result, node, merge_target)` |
| after each `Lattice._record_merge`                     | `dumper.merge_event(layer, new, into, state_hash)`    |
| after each layer's `for` loop completes                | `dumper.layer_end(layer, lattice)`                    |
| at function return (any verdict)                       | `dumper.summary(verdict, stats)`                       |

The hooks are no-ops when `dumper is None`. Existing
backbone tests pass unchanged.
