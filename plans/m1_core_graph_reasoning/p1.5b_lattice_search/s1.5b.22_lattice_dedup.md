### 1. `LatticeProof` data class + supporting types

Fill `inference/monotonic/lattice.py`:

```python
@dataclass(frozen=True)
class LatticeProof:
    """Lattice engine's terminal output — the proof artefact
    per Q1.5b.6. Returned in Ambiguity.proof (gaps_solve) or
    Contradiction.proof (contradictions_solve).
    """
    # Filled by gaps_solve only.
    solutions:        tuple[SolutionRecord, ...]

    # Filled by contradictions_solve only.
    dead_commitments: tuple[DeadCommitment, ...]

    # Filled when store_lattice=True (regardless of entry).
    # Maps state_hash → canonical SetNode.
    kb_index:         dict[int, SetNode]

    # Surviving frontier at termination — alive size-N sets
    # if depth cap hit; empty otherwise.
    alive_at_end:     tuple[CanonicalSetId, ...]

    # Snapshot of root._nogoods at termination.
    learned_nogoods:  frozenset[frozenset[FactId]]

    # Cumulative counters.
    stats:            LatticeStats


@dataclass(frozen=True)
class SolutionRecord:
    """One satisfying commitment + its saturated kb (the only
    persisted per-set kb — alive intermediates are not stored).
    """
    commitment: CanonicalSetId
    kb:         KnowledgeBase   # deep snapshot for P1.6's derivation_dag walk
    firings:    tuple[Firing, ...]
    layer:      int


@dataclass(frozen=True)
class DeadCommitment:
    """One refuted commitment + its unsat-core."""
    commitment:     CanonicalSetId
    unsat_core:     frozenset[Fact]
    learned_clause: frozenset[FactId]
    layer:          int
    kind:           Literal["dead-pre", "dead-post"]


@dataclass(frozen=True)
class SetNode:
    """Cross-set state-hash merge target — populated under
    store_lattice=True. Under gaps_solve, multiple SetNodes
    can share the same state_hash (no merge — per GAPS
    contract); under contradictions_solve they merge.
    """
    state_hash:    int
    canonical_set: CanonicalSetId       # first arrival's commitment
    labels:        tuple[CanonicalSetId, ...]   # all arrivals (multilabel under contradictions)
    verdict:       Literal["alive", "dead", "solution"]
    layer:         int


@dataclass
class LatticeStats:
    enterings_total:       int = 0
    enterings_alive:       int = 0
    enterings_dead_pre:    int = 0
    enterings_dead_post:   int = 0
    solutions_found:       int = 0
    facts_merged:          int = 0
    forced_positives:      int = 0
    saturate_count:        int = 0
    layers_explored:       int = 0
    nogoods_emitted:       int = 0
    nogoods_subsumed:      int = 0
    state_hash_merges:     int = 0
    elapsed_seconds:       float = 0.0
```

### 2. `KnowledgeBase.snapshot()`

New method on `kb/store.py`:

```python
def snapshot(self) -> KnowledgeBase:
    """Deep-ish copy for archival.

    Copies: ``facts`` list, every fact's ``provenance``
    chain, ``_nogoods`` set, ``_negated_facts`` set, all
    fact reverse-indexes.

    Shares by reference: ``types``, ``instances``,
    ``relations``, ``rules``, ``hrules``, ``classes`` —
    these are constant across the search, no mutation
    concern.

    Soundness: a snapshotted KB's ``derivation_dag(f)``
    returns the same chain as the source at snapshot time,
    even if the source mutates later. The shared registries
    don't matter for derivation walking (provenance carries
    the rule reference; we only need name equality).
    """
    new = KnowledgeBase()
    # Share registries by reference.
    new.types        = self.types
    new.instances    = self.instances
    new.relations    = self.relations
    new.rules        = self.rules
    new.hrules       = self.hrules
    new.classes      = self.classes
    new.query        = self.query
    new.config       = self.config
    # Deep-copy mutable state.
    new.facts            = list(self.facts)
    new._nogoods         = set(self._nogoods)
    new._negated_facts   = set(self._negated_facts)
    new.committed_hypotheses = set(self.committed_hypotheses)
    new.rebuild_indexes()
    return new
```

### 3. `store_lattice` flag wiring + state-hash dedup

In `_explore_layers`:

```python
def _explore_layers(root_kb, state, *, store_lattice: bool = False, ...):
    # ... existing setup ...

    for layer in 1..max_set_size:
        for C in candidates:
            # ... existing budget gates ...

            result = try_commitment_set(root_kb, C)

            # State-hash dedup checkpoint (S1.5b.22 §3b).
            # Only when store_lattice=True AND entry is contradictions
            # AND result is post-saturation (skip for dead-pre since
            # we haven't saturated).
            if (
                store_lattice
                and state.entry == "contradictions"
                and result.kind != "dead-pre"
            ):
                h = state_hash(result.kb)
                if h in state.kb_index:
                    existing = state.kb_index[h]
                    state.kb_index[h] = SetNode(
                        state_hash=h,
                        canonical_set=existing.canonical_set,
                        labels=existing.labels + (C,),
                        verdict=existing.verdict,
                        layer=existing.layer,
                    )
                    state.stats.state_hash_merges += 1
                    # SKIP downstream — the existing node has
                    # already done all the flat root-writes
                    # and entry-side collection.
                    continue
                else:
                    state.kb_index[h] = SetNode(
                        state_hash=h,
                        canonical_set=C,
                        labels=(C,),
                        verdict={"alive": "alive", "dead-pre": "dead", "dead-post": "dead"}[result.kind],
                        layer=layer,
                    )

            # Under gaps_solve with store_lattice=True: we DO populate
            # kb_index but never merge — every SetNode stays with
            # labels=(C,). This preserves the GAPS contract that
            # distinct satisfying commitments register separately.
            elif store_lattice and state.entry == "gaps" and result.kind != "dead-pre":
                h = state_hash(result.kb)
                state.kb_index[h] = state.kb_index.get(h)  # no overwrite — append as labels
                # NOTE: under GAPS we KEEP the first arrival's SetNode
                # but record this C in a per-hash list for the dumper;
                # the merge step is what's skipped, not the storage.
                # Implementation: state.kb_index[h] grows labels but
                # the explicit append-without-merge variant —
                # implementation-detail-y; see s1.5b.22 code.

            # ... existing dead / alive handling ...

    # Phase 3 — finalise the proof.
    proof = LatticeProof(
        solutions       = tuple(state.solutions),
        dead_commitments= tuple(state.dead_commitments),
        kb_index        = state.kb_index if store_lattice else {},
        alive_at_end    = A_N_tuple,
        learned_nogoods = frozenset(root_kb._nogoods),
        stats           = state.stats,
    )
    if state.entry == "gaps":
        return Ambiguity(branches=…, unresolved=…, proof=proof), state.stats
    if state.entry == "contradictions":
        cores = frozenset().union(*(d.unsat_core for d in state.dead_commitments))
        return Contradiction(unsat_core=cores, proof=proof), state.stats
```
