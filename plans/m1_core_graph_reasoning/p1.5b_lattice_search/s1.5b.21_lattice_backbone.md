## Goal

Fill `lattice_solve` with a working BFS-by-size exhaustive
search that:

1. Builds a `Lattice` with one `SetNode` per
   canonically-distinct commitment set entered.
2. **No state-hash dedup yet** — every set entered produces
   a fresh SetNode (S1.5b.22 collapses identical states).
3. **No multi-parent integrate** — each set's unconditional
   facts go straight to root (the monotonic pattern, but
   without monotonic's early termination).
4. **No forced-positive mining via integrate** — unconditional
   facts come from `try_set`'s extraction; integrate-style
   per-parent propagation lands in S1.5b.24.
5. **Exhaustive** to `max_set_size` — no early-exit when
   `is_solved(root.kb)` becomes True.
6. Verdict trichotomy from
   [Q1.5b.7](open_questions.md#q15b7--termination--completeness--mode-handling):
   Solution / Ambiguity / Contradiction computed from the
   final lattice frontier.

After this stage:

- `bench_dag.py examples/branching/03_five_hyps_one_alive.ein`
  returns `Solution` with matching bindings to tree +
  monotonic.
- `bench_dag.py examples/zebra2.ein --max-set-size 5` runs
  exhaustively; verdict depends on whether the puzzle is
  uniquely solvable (Solution) or whether monotonic-style
  early termination would have stopped sooner (still
  Solution — lattice just keeps going past it).

## Algorithm

```
lattice_solve(root_kb, max_set_size, config, mode=SOLVE):
    Saturator(root_kb).saturate()
    if not generate_alive(root_kb):
        return Contradiction()

    lattice = Lattice()
    alive = compute_alive(root_kb)
    A_prev = layer_1(alive)

    # Allocate layer-0 root SetNode (empty commitment).
    root_node = SetNode(
        canonical_set=(),
        state_hash=state_hash(root_kb),
        kb_snapshot=root_kb,
        verdict="alive",
    )
    lattice.add(root_node)

    for layer in range(1, max_set_size + 1):
        if layer == 1:
            candidates = A_prev
        else:
            candidates = generate_layer(
                A_prev, alive=alive, nogoods=root_kb._nogoods,
            )
        candidates.sort()   # lex; scoring in S1.5b.26

        A_layer: list[CanonicalSetId] = []
        for C in candidates:
            result = try_set(root_kb, C)

            node = SetNode(
                canonical_set=C,
                state_hash=state_hash(result.kb),
                kb_snapshot=result.kb,
                firings=result.firings,
                verdict=_classify_verdict(result, mode),
                # S1.5b.22 fills labels + parents; the
                # backbone leaves them as singletons.
                labels=(C,),
                parents=tuple(_parent_sets(C, A_prev)),
                # S1.5b.25 fills these via integrate; for
                # the backbone we copy result's
                # unconditional facts directly.
                forced_positives=result.unconditional_facts,
                learned_nogoods=(
                    (frozenset(C),) if result.kind != "alive" else ()
                ),
            )
            lattice.add(node)

            if result.kind in ("dead-pre", "dead-post"):
                emit_nogood(root_kb, frozenset(C))
                if len(C) == 1:
                    _emit_negated_fact_writeback(root_kb, C[0])
                # NB: no early-exit; lattice exhausts.
                continue

            # Alive — merge unconditional facts into root.
            # Backbone uses the monotonic pattern (merge to
            # root directly). S1.5b.24 replaces this with
            # multi-parent integrate.
            root_changed = False
            for f in result.unconditional_facts:
                if root_kb._fact_by_id(f.relation_name, f.args) is None:
                    stored = root_kb.add_fact(f)
                    root_kb._index_fact(stored)
                    root_changed = True

            if root_changed:
                Saturator(root_kb).saturate()
                alive = compute_alive(root_kb)
                # Lattice keeps exploring even after root
                # gains the goal — exhaustive search.

            if _is_solved(result.kb, mode):
                node.verdict = "solution"

            A_layer.append(C)

        if not A_layer:
            break
        A_prev = A_layer

    return _synthesise_verdict(lattice, mode)

# ── Helpers ───────────────────────────────────────────────

def _classify_verdict(result, mode):
    if result.kind in ("dead-pre", "dead-post"):
        return "dead"
    # Will be promoted to "solution" if _is_solved fires.
    return "alive"

def _parent_sets(C, A_prev):
    """All (|C|-1)-subsets of C that are in A_prev. For
    layer 1, the only parent is the root SetNode (empty)."""
    if len(C) == 1:
        return ((),)
    return tuple(
        T for T in [tuple(x for x in C if x != C[i])
                    for i in range(len(C))]
        if T in set(A_prev)
    )

def _synthesise_verdict(lattice, mode) -> Verdict:
    """Compute Solution / Ambiguity / Contradiction from
    the final lattice state. See Q1.5b.7 trichotomy."""
    solutions = [
        n for n in lattice.nodes_by_set.values()
        if n.verdict == "solution"
    ]
    alive_at_max = [
        n for n in lattice.nodes_by_set.values()
        if n.verdict == "alive" and
           len(n.canonical_set) >= max_set_size_actual
    ]
    if len(solutions) == 1 and not alive_at_max:
        return Solution(kb=solutions[0].kb_snapshot, trace=(), tree=None)
    if not solutions and not alive_at_max:
        return Contradiction(unsat_core=frozenset(), tree=None)
    return Ambiguity(
        branches=tuple(
            Solution(kb=n.kb_snapshot, trace=(), tree=None)
            for n in solutions
        ),
        unresolved=tuple(),  # S1.5b.29 fills via LatticeProof
        tree=None,
    )
```

The `_synthesise_verdict` is the trichotomy-from-Q1.5b.7
applied: exactly-one solution + no alive-at-max → Solved;
no solutions + no alive-at-max → Contradiction; otherwise
Ambiguity.

```python
def add(self, node: SetNode) -> None:
    """Register a SetNode in the lattice. Backbone: no
    dedup check; S1.5b.22 makes this conditional on
    state-hash + canonical-set lookup."""
    self.nodes_by_set[node.canonical_set] = node
    self.nodes_by_hash[node.state_hash] = node
    layer = len(node.canonical_set)
    while len(self.alive_by_layer) <= layer:
        self.alive_by_layer.append([])
        self.dead_by_layer.append([])
    if node.verdict in ("alive", "solution"):
        self.alive_by_layer[layer].append(node.canonical_set)
    else:
        self.dead_by_layer[layer].append(node.canonical_set)
```
