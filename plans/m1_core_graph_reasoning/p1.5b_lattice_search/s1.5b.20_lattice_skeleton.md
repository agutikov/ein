### `bench_lattice.py` (skeleton)

```python
"""Lattice engine CLI — dispatches to gaps_solve or
contradictions_solve based on flags.

Usage:
    bench_lattice.py --gaps puzzle.ein
    bench_lattice.py --contradictions puzzle.ein
    bench_lattice.py --gaps --store-lattice --dump-states /tmp/d puzzle.ein
"""
import argparse
import sys
from pathlib import Path

from ein_bot.inference.lattice import gaps_solve, contradictions_solve
from ein_bot.inference.lattice.state_dump import LatticeDumper
from ein_bot.kb.from_ir import load_file


def main() -> int:
    p = argparse.ArgumentParser(...)
    mode = p.add_mutually_exclusive_group(required=True)
    mode.add_argument("--gaps", action="store_true",
                       help="gaps_solve: enumerate all solutions")
    mode.add_argument("--contradictions", action="store_true",
                       help="contradictions_solve: enumerate all deaths")
    p.add_argument("--store-lattice", action="store_true",
                    help="opt-in per-SetNode storage")
    p.add_argument("--max-set-size", type=int, default=5)
    p.add_argument("--dump-states", type=Path, default=None)
    p.add_argument("--max-time", type=float, default=None)
    p.add_argument("--max-enterings", type=int, default=None)
    p.add_argument("puzzle", type=Path)
    args = p.parse_args()

    kb = load_file(args.puzzle)
    dumper = LatticeDumper(str(args.dump_states)) if args.dump_states else None
    entry = gaps_solve if args.gaps else contradictions_solve
    verdict, stats = entry(
        kb,
        max_set_size=args.max_set_size,
        store_lattice=args.store_lattice,
        dumper=dumper,
        max_time=args.max_time,
        max_enterings=args.max_enterings,
    )
    # Print verdict + stats; skip details until backbones land.
    print(verdict)
    print(stats)
    return 0


if __name__ == "__main__":
    sys.exit(main())
```
