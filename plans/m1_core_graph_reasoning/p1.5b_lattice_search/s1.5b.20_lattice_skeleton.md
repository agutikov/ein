"""SetNode + Lattice — the lattice's data structures.

Stub — backbone (sans dedup/multilabel) in S1.5b.21;
multilabel + multi-parent storage in S1.5b.22.
"""
from __future__ import annotations
from dataclasses import dataclass, field
from typing import Any

from ein_bot.inference.apriori import CanonicalSetId, FactId
from ein_bot.kb.store import KnowledgeBase

@dataclass
class SetNode:
    """A node in the search lattice — one canonical set per
    state-deduped commitment.

    Stub fields; the full surface from
    [Q1.5b.4](../../../plans/m1_core_graph_reasoning/p1.5b_lattice_search/open_questions.md#q15b4--set-equivalence-dedup--state-hash-dedup)
    lands in S1.5b.22 (multilabel + multi-parent).
    """
    canonical_set:  CanonicalSetId
    state_hash:     int = 0
    kb_snapshot:    Any = None                          # KnowledgeBase | None
    firings:        tuple = ()
    verdict:        str = "alive"                       # Literal["alive","dead","solution","open"]
    # ── Filled in S1.5b.22 (dedup) ─────────────────────
    labels:         tuple[CanonicalSetId, ...] = ()
    parents:        tuple[CanonicalSetId, ...] = ()
    # ── Filled in S1.5b.25 (forced positives) ──────────
    proposed_negatives:  tuple = ()                     # tuple[Fact, ...]
    forced_positives:    tuple = ()
    learned_nogoods:     tuple = ()                     # tuple[frozenset[FactId], ...]

@dataclass
class Lattice:
    """The search-lattice store — index of SetNodes by
    canonical_set AND by state_hash.

    Stub — backbone in S1.5b.21; secondary state-hash index
    in S1.5b.22.
    """
    nodes_by_set:    dict[CanonicalSetId, SetNode] = field(default_factory=dict)
    nodes_by_hash:   dict[int, SetNode] = field(default_factory=dict)
    # Per-layer alive set IDs:
    alive_by_layer:  list[list[CanonicalSetId]] = field(default_factory=list)
    dead_by_layer:   list[list[CanonicalSetId]] = field(default_factory=list)

    def get_by_set(self, canonical_set: CanonicalSetId) -> SetNode | None:
        return self.nodes_by_set.get(canonical_set)

    def get_by_hash(self, state_hash: int) -> SetNode | None:
        return self.nodes_by_hash.get(state_hash)

## CLI entry point — `bench_dag.py`

`ein.py/demo/bench_dag.py`:

```python
"""Run the lattice set-search engine on a .ein file.

Mirrors `bench_solve.py` / `bench_monotonic.py` CLI shape;
reaches into `inference/lattice/`. Output: verdict + per-layer
summary + (if --dump-states) full LatticeDumper output.

Stub — backbone wired in S1.5b.21.
"""
from __future__ import annotations

import argparse
import sys
from pathlib import Path

sys.path.insert(
    0, str(Path(__file__).resolve().parents[1] / "src"),
)

from ein_bot.inference.config import SolverConfig                  # noqa: E402
from ein_bot.inference.lattice.solver import lattice_solve              # noqa: E402
from ein_bot.inference.lattice.state_dump import LatticeDumper          # noqa: E402
from ein_bot.ir import parse                                         # noqa: E402
from ein_bot.kb.store import KnowledgeBase                           # noqa: E402

def _build_argparser() -> argparse.ArgumentParser:
    ap = argparse.ArgumentParser(description=__doc__.split("\n",1)[0])
    ap.add_argument("puzzle", type=Path)
    ap.add_argument("--max-set-size", type=int, default=5)
    ap.add_argument("--dump-states", type=Path, default=None)
    ap.add_argument("--max-time", type=float, default=None)
    ap.add_argument("--verbose", "-v", action="store_true")
    ap.add_argument("--lattice-order", choices=["lex", "score-sum"],
                    default="score-sum",
                    help="within-layer ordering — Q1.5b.2.c")
    ap.add_argument("--lattice-sanity-check", action="store_true",
                    help="enable saturation-commutativity sanity check (S1.5b.27)")
    return ap

def main(argv: list[str] | None = None) -> int:
    args = _build_argparser().parse_args(argv)
    text = args.puzzle.read_text()
    kb = KnowledgeBase.from_ir(parse(text))
    cfg = SolverConfig()
    # NOTE — S1.5b.21 wires the cfg + dumper + sanity-check.
    verdict = lattice_solve(kb, max_set_size=args.max_set_size,
                            config=cfg)
    print(f"verdict: {type(verdict).__name__}")
    return 0

if __name__ == "__main__":
    sys.exit(main())
```
